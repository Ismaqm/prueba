82  	public static int mantenibilidad(ArrayList<RangosMantenibilidad> tabla,int modularidad, int reusabilidad,int analizabilidad, int modificado, int probado) {
83  		
84          
85          for(int i=0; i<tabla.size();i++) {
86          	if(tabla.get(i).getMinimo() <= modularidad && modularidad <= tabla.get(i).getMaximo()) {
87          		modularidad = tabla.get(i).getModularidad();
88          		break;
89          	}
90          }
91          
92          for(int i=0; i<tabla.size();i++) {
93          	if(tabla.get(i).getMinimo() <= reusabilidad && reusabilidad <= tabla.get(i).getMaximo()) {
94          		reusabilidad = tabla.get(i).getReusabilidad();
95          		break;
96          	}
97          }
98          
99          for(int i=0; i<tabla.size();i++) {
100         	if(tabla.get(i).getMinimo() <= analizabilidad && analizabilidad <= tabla.get(i).getMaximo()) {
101         		analizabilidad = tabla.get(i).getAnalizabilidad();
102         		break;
103         	}
104         }
105         
106         for(int i=0; i<tabla.size();i++) {
107         	if(tabla.get(i).getMinimo() <= modificado && modificado <= tabla.get(i).getMaximo()) {
108         		modificado = tabla.get(i).getCapacidadModi();
109         		break;
110         	}
111         }
112         
113         for(int i=0; i<tabla.size();i++) {
114         	if(tabla.get(i).getMinimo() <= probado && probado <= tabla.get(i).getMaximo()) {
115         		probado = tabla.get(i).getCapacidadProba();
116         		break;
117         	}
118         }
119         //12 --> 1   58-->2   87-->3  34-->1  95-->4
120         //System.out.print(modularidad + " " + reusabilidad + " " + analizabilidad + " " + modificado + " " + probado);
121         
122         int menor =Math.min(modularidad,Math.min(reusabilidad,Math.min(analizabilidad,Math.min(modificado,probado))));
123         System.out.println("El nivel de mantenibilidad es : " + menor);
124         return menor;
125 	}

127 	public static int adecuacionFinal(ArrayList<RangosAdecuaccion> tabla, int completitud, int correccion, int pertinencia) {
128 		
129         for(int i=0; i<tabla.size();i++) {
130         	if(tabla.get(i).getMinimo() <= completitud && completitud <= tabla.get(i).getMaximo()) {
131         		completitud = tabla.get(i).getCompletitud();
132         		break;
133         	}
134         }
135         
136         for(int i=0; i<tabla.size();i++) {
137         	if(tabla.get(i).getMinimo() <= correccion && correccion <= tabla.get(i).getMaximo()) {
138         		correccion = tabla.get(i).getCorrecion();
139         		break;
140         	}
141         }
142         
143         for(int i=0; i<tabla.size();i++) {
144         	if(tabla.get(i).getMinimo() <= pertinencia && tabla.get(i).getMaximo() >= pertinencia ) {
145         		pertinencia = tabla.get(i).getPertinencia();
146         		break;
147         	}
148         }
149         
150      // System.out.println(completitud + " " + correccion + " "  + pertinencia);
151         
152         int menor = completitud;
153         
154         if (correccion < pertinencia) {
155         	if(correccion<completitud) menor = correccion;
156         } else { if(pertinencia<completitud) menor = pertinencia; }
157         
158        // System.out.println("La adecuacion final es: " + menor);
159         
160         return menor;
161         
162 	}
